1//! Sorting algorithms implementation
2//! 
3//! Currently provides quick sort implementation for i32 arrays.

5/// Internal helper function for quick sort
6/// Partitions the array and returns the pivot index
7fn partition(array: &mut [i32], low: usize, high: usize) -> usize {
8    let pivot = array[low];
9    let (mut left, mut right) = (low, high);
10
11    while left < right {
12        while left < right && array[right] >= pivot {
13            right -= 1;
14        }
15        array[left] = array[right];
16
17        while left < right && array[left] <= pivot {
18            left += 1;
19        }
20        array[right] = array[left];
21    }
22
23    array[left] = pivot;
24    left
25}
26
27/// Sorts an i32 array in-place using quick sort algorithm
28///
29/// # Arguments
30/// * `array` - Mutable slice of i32 to be sorted
31/// * `low` - Starting index of the range to sort (inclusive)
32/// * `high` - Ending index of the range to sort (inclusive)
33///
34/// # Examples
35/// ```
36/// use data_structure::sorting::quick;
37///
38/// let mut arr = [5, 2, 4, 1, 3];
39/// quick(&mut arr, 0, 4);
40/// assert_eq!(arr, [1, 2, 3, 4, 5]);
41/// ```
42pub fn quick(array: &mut [i32], low: usize, high: usize) {
43    if low < high {
44        let pivot_loc = partition(array, low, high);
45        quick(array, low, pivot_loc.saturating_sub(1));
46        quick(array, pivot_loc + 1, high);
47    }
48}
