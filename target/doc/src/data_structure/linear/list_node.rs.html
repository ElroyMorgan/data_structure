1/// 链表节点结构体
2///
3/// # 示例
4/// ```
5/// use data_structure::linear::list_node::ListNode;
6/// let mut head = ListNode::<i32>::new();
7/// head.push(1);
8/// head.push(2);
9/// ```
10pub struct ListNode<T> {
11    /// 节点存储的数据，哨兵节点为None
12    pub data: Option<T>,
13    /// 指向下一个节点的指针
14    pub next: Option<Box<ListNode<T>>>,
15}
16
17impl<T> ListNode<T> {
18    /// 创建一个新的哨兵节点(不包含数据)
19    ///
20    /// # 示例
21    /// ```
22    /// use data_structure::linear::list_node::ListNode;
23    /// let sentinel = ListNode::<i32>::new();
24    /// assert!(sentinel.data.is_none());
25    /// ```
26    pub fn new() -> Self {
27        ListNode {
28            data: None,
29            next: None,
30        }
31    }
32
33    /// 获取指定索引(0-based)节点的不可变引用
34    /// 如果索引越界则返回None
35    ///
36    /// # 示例
37    /// ```
38    /// use data_structure::linear::list_node::ListNode;
39    /// let mut head = ListNode::new();
40    /// head.push(42);
41    /// assert_eq!(head.get(1).unwrap().data, Some(42));
42    /// ```
43    pub fn get(&self, index: usize) -> Option<&Self> {
44        let mut current = self;
45        for _ in 0..index {
46            current = current.next.as_ref()?;
47        }
48        Some(current)
49    }
50
51    /// 获取指定索引(0-based)节点的可变引用
52    /// 如果索引越界则返回None
53    ///
54    /// # 示例
55    /// ```
56    /// use data_structure::linear::list_node::ListNode;
57    /// let mut head = ListNode::new();
58    /// head.push(42);
59    /// *head.get_mut(1).unwrap().data.as_mut().unwrap() = 24;
60    /// ```
61    pub fn get_mut(&mut self, index: usize) -> Option<&mut Self> {
62        let mut current = self;
63        let mut count = 0;
64        // 边界检查：当 index 为 0 时直接返回当前节点
65        if index == 0 {
66            return Some(current);
67        }
68        while count < index - 1 {
69            // 使用 as_mut()? 自动处理 None 情况
70            current = current.next.as_mut()?;
71            count += 1;
72        }
73        // 最后一次移动需要特别检查
74        current.next.as_mut().map(|node| &mut **node)
75    }
76
77    /// 删除链表尾部节点
78    ///
79    /// # 示例
80    /// ```
81    /// use data_structure::linear::list_node::ListNode;
82    /// let mut head = ListNode::new();
83    /// head.push(1);
84    /// head.push(2);
85    /// head.pop_tail();
86    /// assert_eq!(head.length(), 1);
87    /// ```
88    pub fn pop_tail(&mut self) {
89        if self.next.is_none() {
90            return;
91        }
92        
93        let mut current = self;
94        while current.next.as_ref().unwrap().next.is_some() {
95            current = current.next.as_mut().unwrap();
96        }
97        current.next = None;
98    }
99
100    /// 在链表尾部添加新节点
101    ///
102    /// # 示例
103    /// ```
104    /// use data_structure::linear::list_node::ListNode;
105    /// let mut head = ListNode::new();
106    /// head.push(42);
107    /// assert!(head.next.is_some());
108    /// ```
109    pub fn push(&mut self, data: T) {
110        let mut tail = self;
111        while tail.next.is_some() {
112            tail = tail.next.as_mut().unwrap();
113        }
114        tail.next = Some(Box::new(ListNode {
115            data: Some(data),
116            next: None,
117        }));
118    }
119
120    /// 在指定位置插入新节点
121    ///
122    /// # 示例
123    /// ```
124    /// use data_structure::linear::list_node::ListNode;
125    /// let mut head = ListNode::new();
126    /// head.push(1);
127    /// head.push(3);
128    /// head.insert(1, 2);
129    /// assert_eq!(head.get(1).unwrap().data, Some(2));
130    /// ```
131    pub fn insert(&mut self, index: usize, data: T){
132        let current= self.get_mut(index-1).unwrap();
133        let new_node=Box::new(ListNode{
134           data:Some(data), 
135            //take临时取出current.next的所有权
136            next:current.next.take(),
137        });
138         current.next=Some(new_node);
139    }
140
141    /// 删除指定位置的节点
142    ///
143    /// # 示例
144    /// ```
145    /// use data_structure::linear::list_node::ListNode;
146    /// let mut head = ListNode::new();
147    /// head.push(1);
148    /// head.push(2);
149    /// head.remove(1);
150    /// assert_eq!(head.length(), 1);
151    /// ```
152    pub fn remove(&mut self, index: usize)  {
153        let current = self.get_mut(index - 1).unwrap();
154        let removed_node = current.next.take().unwrap();
155        current.next = removed_node.next;
156    }
157
158    /// 获取链表长度(不包含哨兵节点)
159    ///
160    /// # 示例
161    /// ```
162    /// use data_structure::linear::list_node::ListNode;
163    /// let mut head = ListNode::new();
164    /// head.push(1);
165    /// head.push(2);
166    /// assert_eq!(head.length(), 2);
167    /// ```
168    pub fn length(&mut self)-> usize{
169        let mut count:usize = 0;
170        let mut current = self;
171        while current.next.is_some(){
172            count+=1;
173            current=current.next.as_mut().unwrap();
174        }
175        count
176    }
177}
